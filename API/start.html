# 快速上手

引入 funt-ui

1.引入所有组件

import Vue from 'vue';
import Funt from 'funt-ui';
import 'funt-ui/lib/styles/funt-ui.css';
Vue.use(Funt);

2.按需引入组件

import Vue from 'vue';
import { Loading, Scroll } from 'funt-ui';
import 'funt-ui/lib/styles/funt-ui.css';
Vue.use(Loading);
Vue.use(Scroll);
